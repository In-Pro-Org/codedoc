4linkfunction ConvertTo-Hierarchical {

5link    param($data)

6link

7link    $root = [Ordered]@{

8link        name     = "Root"

9link        children = @()

10link    }

11link

12link    foreach ($item in $data) {

13link        $namespaceNode = [Ordered]@{

14link            name     = $item.Name

15link            children = @()

16link        }

17link

18link        foreach ($obj in $item.Objects) {

19link            $namespaceNode.children += [Ordered]@{

20link                name = $obj.Name

21link            }

22link        }

23link

24link        $root.children += $namespaceNode

25link    }

26link

27link    return $root

28link}

190linkfunction Get-GuiClasses {

191link

192link    $AllClasses = [System.AppDomain]::CurrentDomain.GetAssemblies() | 

193link        Foreach-Object { $_.GetExportedTypes() } | 

194link            Select-Object Name, Fullname, Namespace, BaseType, ReflectedType, MemberType, Module, TypeHandle, UnderlyingSystemType, Assembly

195link

196link    $GuiClasses = $AllClasses | 

197link        Select-Object -Property Namespace -Unique | 

198link            Where-Object { $_.Namespace -match 'MahApps|MaterialDesign|ControlzEx|System.Xaml|Microsoft.Xaml|LiveCharts|LoadingIndicators' } | 

199link                Sort-Object -Property Namespace

200link

201link    $ClassDict = $GuiClasses | Foreach-Object {

202link        $currentNamespace = $_.Namespace

203link        $nestedObjects = $AllClasses | Where-Object { $_.Namespace -eq $currentNamespace }

204link

205link        [PSCustomObject]@{

206link            Name    = $currentNamespace

207link            Objects = $nestedObjects

208link        }

209link    }

210link

211link

212link    $hierarchicalData = ConvertTo-Hierarchical -data $ClassDict

213link    $jsonData = $hierarchicalData | ConvertTo-Json -Depth 3

214link    Build-HTMLTree -jsonData $jsonData

215link

216link    $json = $ClassDict | ConvertTo-Json -Depth 50 | Out-File -FilePath 'GuiClasses.json' -Encoding utf8

217link

218link    $jsonData = $ClassDict | ForEach-Object {

219link        @{

220link            id     = $_.Name

221link            parent = "#"

222link            #text   = $_.Name

223link        },

224link        ($_.Objects | ForEach-Object {

225link            @{

226link                id     = $_.FullName

227link                parent = $_.Namespace

228link                text   = $_.Name

229link            }

230link        })

231link    } | ConvertTo-Json -Compress

232link    ###############################################################

233link    $children = @()

234link    $jsonData = $ClassDict | ForEach-Object {

235link

236link        $_.Objects | ForEach-Object {

237link            $children += @{

238link                id     = $_.FullName

239link                parent = $_.Namespace

240link                text   = $_.Name

241link            }

242link        }

243link

244link        @{

245link            id       = $_.Name

246link            parent   = "#"

247link            text     = $_.Name

248link            children = $children

249link

250link        }

251link    } | ConvertTo-Json -Depth 50 -Compress

252link

253link    ###############################################################

254link    $flatData = @()

255link    $ClassDict | ForEach-Object {

256link        $flatData += @{

257link            id     = $_.Name

258link            parent = "#"

259link            text   = $_.Name

260link        }

261link        $_.Objects | ForEach-Object {

262link            $flatData += @{

263link                id     = $_.FullName

264link                parent = $_.Namespace

265link                text   = $_.Name

266link            }

267link        }

268link    }

280link<body>

281link    <div id="jstree"></div>

282link

283link    <script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/3.6.0/jquery.min.js"></script>
