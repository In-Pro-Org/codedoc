35link'Installing PowerShell Modules'

36linkforeach ($module in $modulesToInstall) {

37link    $installSplat = @{

38link        Name               = $module.ModuleName

39link        RequiredVersion    = $module.ModuleVersion

40link        Repository         = 'PSGallery'

41link        SkipPublisherCheck = $true

42link        Force              = $true

43link        ErrorAction        = 'Stop'

44link    }

45link    try {

46link        Install-Module @installSplat

47link        Import-Module -Name $module.ModuleName -ErrorAction Stop

48link        '  - Successfully installed {0}' -f $module.ModuleName

49link    }

50link    catch {
